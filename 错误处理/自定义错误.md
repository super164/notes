自定义错误：
需要调用errors包下的new函数，然后返回error类型
```go
package main

import (
	"errors"
	"fmt"
)

func main() {
	err := test()
	if err != nil {
		fmt.Println("自定义的错误：", err)
        
	}
	fmt.Println("上面的除法正常执行了")
	fmt.Println("正常执行下面的逻辑")
}
func test() (err error) {
	num1 := 2
	num2 := 0
	if num2 == 0 {
		//抛出自定义的错误：
		return errors.New("除数不能为0")
	} else {
		result := num1 / num2
		fmt.Println(result)
		return nil
	}

}

```

>如果想要出现错误，处理错误后不继续执行代码，实现终止的操作可以调用`panic(error)`，来实现终止其后面的代码运行
```go
func main() {
	err := test()
	if err != nil {
		fmt.Println("自定义的错误：", err)
		panic(err)
	}
	fmt.Println("上面的除法正常执行了")
	fmt.Println("正常执行下面的逻辑")
}
```
